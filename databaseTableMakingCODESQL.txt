CREATE TABLE PURCHASEORDERLINE (
LINEID int GENERATED ALWAYS AS IDENTITY NOT NULL PRIMARY KEY,
PURCHASEID INT NOT NULL,
PARTID INT NOT NULL,
QUANTITY INT NOT NULL,
PRICEPERUNIT DECIMAL(6,2) NOT NULL,
SUPPLIER VARCHAR(40) NOT NULL,
FOREIGN KEY  (PARTID) REFERENCES STOREROOM(PARTID),
FOREIGN KEY  (PURCHASEID) REFERENCES PURCHASEORDERS(PURCHASEID)
);

CREATE TABLE STOREROOM(
PARTID INT GENERATED ALWAYS AS IDENTITY NOT NULL PRIMARY KEY,
MANUFACTURER VARCHAR(40) NOT NULL,
MANUFACTURERPARTNUMBER VARCHAR(40) NOT NULL,
KANBANSIZE INT NOT NULL,
SAFETYLEVEL INT NOT NULL
);

CREATE TABLE PURCHASEORDERS (
PURCHASEID INT GENERATED ALWAYS AS IDENTITY NOT NULL PRIMARY KEY,
DEPARTMENTCODE VARCHAR(40) NOT NULL,
STATUS VARCHAR(40) NOT NULL,
DELIVERYATTENTION VARCHAR(40) NOT NULL,
COMPLETEDSTATUS BOOLEAN NOT NULL
);
